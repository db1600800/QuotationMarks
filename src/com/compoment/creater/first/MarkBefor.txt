public String is20Length(String value) {
		int chineseCount = getChineseCount(value);
		if (chineseCount>0) {

			if(chineseCount*2>=26)
			{
				chineseCount=26/2;
				value=(String) value.subSequence(0, 13);
			}else
			{
				int aBCCount=value.length()-chineseCount;

				for (int j = 0; j < 26 - chineseCount*2-aBCCount; j++) {
					value += " ";
				}
			}

		} else {
			int length = value.length();
			if (length < 26) {
				for (int j = 0; j < 26 - length; j++) {
					value += " ";
				}
			} else {
				value = value.substring(0, 26);
			}
		}
		return value;
	}


	public static int getChineseCount(String content) {

		int count=0;
		char []cs=content.toCharArray();
		for (int i = 0; i < cs.length; i++) {
			char c = cs[i];
			if(isChinese(c))
			{
				count++;
			}
		}
		return count;
	}

	  public static boolean isChinese(char c) {

	        Character.UnicodeBlock ub = Character.UnicodeBlock.of(c);

	        if (ub == Character.UnicodeBlock.CJK_UNIFIED_IDEOGRAPHS

	                || ub == Character.UnicodeBlock.CJK_COMPATIBILITY_IDEOGRAPHS

	                || ub == Character.UnicodeBlock.CJK_UNIFIED_IDEOGRAPHS_EXTENSION_A

	                || ub == Character.UnicodeBlock.GENERAL_PUNCTUATION

	                || ub == Character.UnicodeBlock.CJK_SYMBOLS_AND_PUNCTUATION

	                || ub == Character.UnicodeBlock.HALFWIDTH_AND_FULLWIDTH_FORMS ) {

	            return true;

	        }

	        return false;

	    }
