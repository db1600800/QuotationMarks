package com.chinapost.palmpost;

import android.content.Context;
import android.util.AttributeSet;
import android.view.MotionEvent;
import android.widget.HorizontalScrollView;

public class DeliveryRateCHScrollView extends HorizontalScrollView {
	
	DeliveryRateActivity activity;
	
	public DeliveryRateCHScrollView(Context context, AttributeSet attrs, int defStyle) {
		super(context, attrs, defStyle);
		activity = (DeliveryRateActivity) context;
	}

	
	public DeliveryRateCHScrollView(Context context, AttributeSet attrs) {
		super(context, attrs);
		if(context!=null && context instanceof DeliveryRateActivity)
		activity = (DeliveryRateActivity) context;
	}

	public DeliveryRateCHScrollView(Context context) {
		super(context);
		activity = (DeliveryRateActivity) context;
	}
	
	@Override
	public boolean onTouchEvent(MotionEvent ev) {
		activity.mTouchView = this;
		return super.onTouchEvent(ev);
	}
	
	@Override
	protected void onScrollChanged(int l, int t, int oldl, int oldt) {
	
		if(activity.mTouchView == this) {
			activity.onScrollChanged(l, t, oldl, oldt);
		}else{
			super.onScrollChanged(l, t, oldl, oldt);
		}
	}
}
